系统地址：

```sql
select * from t_user t where t.ARCHIVE_FLAG = '0';
select * from T_ROLE t;
SELECT t1.uniform_id,
RTRIM(XMLAGG(XMLELEMENT(e, t.role_id || ',') ORDER BY
t.role_id).EXTRACT('//text()'), ',') AS role_ids
FROM T_USER_ROLE t
INNER JOIN t_user t1 ON t.user_id = t1.proxyid
WHERE t1.uniform_id IS NOT NULL
GROUP BY t1.uniform_id;
-- 删除
select * from T_USER_ROLE t where t.user_id in (select proxyid from t_user t
where t.ARCHIVE_FLAG = '0' and t.uniform_id in ('LINQIUYAN', 'LIUJING19')) for
update;
```
