修改的视图： V_OILFEE_INVOICE_DETAIL

```sql
create or replace force view "V_OILFEE_INVOICE_DETAIL"
("GL_DATE","FLIGHT_ID","FLIGHT_TYEP","FLIGHT_DATE","FLIGHT_NO","COMPANY_NAME","DE
PT_AIRPORT","ARRI_AIRPORT","FEE_NAME","ACTYPE","TAIL_NO","QTY_KG","PRICE","AMOUNT
","CURRENCY","TAX_AMOUNT","NO_TAX_AMOUNT","ACCRUED_AMOUNT","CREATE_TIME","SUPPLIE
R_NUMBER","RECEIPT_NO","FLTID","REFUELVOLUME","VOLUME_MEASURE_UNIT","DENSITY","TE
MPERATURE","TEMPERATURE_MEASURE_UNIT","SBCPERCENT","REFUELTYPE","RECEIPTTYPE","RE
FUELING_DATE","STARTREFUELTIME","ENDREFUELTIME")
as SELECT
a.gl_date,
o.flight_id,
CASE
WHEN d.root = '1000' THEN 'DOME'
WHEN d.root = '2000' THEN 'INTL'
WHEN d.root IN ('3000', '4000', '5000') THEN 'AREA'
END FLIGHT_TYEP,
r.refueling_date flight_date,
r.flight_no,
c.company_name,
r.dept_airport,
r.arri_airport,
'OILFEE' fee_name,
d.actype,
r.tail_no,
CASE
WHEN r.density IS NOT NULL THEN r.qty_kg
ELSE r.qty * DECODE(
r.measure_unit,
'TON', 1000,
exchange_unit(r.measure_unit, 'KG', r.refueling_date, 'MEASURE')
)
END AS qty_kg,
CASE
WHEN r.QTY = 0 THEN 0
ELSE ROUND(
(r.amount + NVL(r.share_surcharge_amount, 0)) /
NULLIF(
CASE
WHEN r.density IS NOT NULL THEN r.qty_kg
ELSE r.qty * DECODE(
r.measure_unit,
'TON', 1000,
exchange_unit(r.measure_unit, 'KG', r.refueling_date,
'MEASURE')
)
END,
), 340
)
END price,
(r.amount + NVL(r.share_surcharge_amount, 0)) amount,
r.currency,
d.tax_amount,
d.no_tax_amount,
d.accrued_amount,
s.create_time,
a.ebs_supplier_id supplier_number,
r.RECEIPT_NO,
t4.fltid,
t3.refuelvolume,
CASE
WHEN t3.refuelvolume IS NULL THEN NULL
ELSE 'L'
END AS volume_measure_unit,
t3.density,
t3.temperature,
CASE
WHEN t3.temperature IS NULL THEN NULL
ELSE '℃'
END AS temperature_measure_unit,
t3.sbcpercent,
CASE
WHEN r.AMOUNT > 0 THEN '0'
ELSE '1'
END AS refueltype,
r.RECEIPT_TYPE receiptType,
r.refueling_date,
t3.startrefueltime,
t3.endrefueltime
FROM
t_fuel_invoice_ap_invoice a
JOIN t_fuel_invoice ii ON ii.ap_invoice_id = a.proxyid
JOIN t_fuel_invoice_receipt r ON ii.proxyid = r.invoice_no
JOIN t_fuel_invoice_write_off o ON r.proxyid = o.receipt_id AND a.proxyid =
o.ap_invoice_id
```
