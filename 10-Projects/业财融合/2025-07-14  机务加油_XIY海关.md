新建机务订单表

```sql
CREATE TABLE "FOFAS"."T_AMFUEL_ORDER"
( "ORDER_ID" NUMBER(16,0) NOT NULL ENABLE,
"AIRCRAFT_NO" VARCHAR2(50) NOT NULL ENABLE,
"FLIGHT_DATE" VARCHAR2(10) NOT NULL ENABLE,
"FLIGHT_NO" VARCHAR2(20) NOT NULL ENABLE,
"AIRPORT" VARCHAR2(50) NOT NULL ENABLE,
"PLANNED_FUEL" NUMBER(15,2) NOT NULL ENABLE,
"PLANNED_FUEL_TIME" VARCHAR2(19) NOT NULL ENABLE,
"AIRCRAFT_TYPE" VARCHAR2(50) NOT NULL ENABLE,
"FUEL_TYPE" VARCHAR2(20) NOT NULL ENABLE,
"CREATOR" VARCHAR2(50) NOT NULL ENABLE,
"CREATOR_CODE" VARCHAR2(50) NOT NULL ENABLE,
"CREATOR_PHONE" VARCHAR2(20) NOT NULL ENABLE,
"SCENE_CONTACTS" VARCHAR2(50) NOT NULL ENABLE,
"SCENE_CONTACTS_PHONE" VARCHAR2(20) NOT NULL ENABLE,
"POS" VARCHAR2(100),
"CREATE_TIME" TIMESTAMP (6) NOT NULL ENABLE,
"STATUS" VARCHAR2(20) NOT NULL ENABLE,
"SYNC_STATUS" VARCHAR2(1) DEFAULT 0 NOT NULL ENABLE,
CONSTRAINT PK_T_AMFUEL_ORDER PRIMARY KEY (ORDER_ID)
);
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.ORDER_ID IS ' 加油订单 ID';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.AIRCRAFT_NO IS ' 机号 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.FLIGHT_DATE IS ' 航班⽇期 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.FLIGHT_NO IS ' 航班号 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.AIRPORT IS ' 加油场站 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.PLANNED_FUEL IS ' 计划加油量 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.PLANNED_FUEL_TIME IS ' 计划加油时间 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.AIRCRAFT_TYPE IS ' 机型 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.FUEL_TYPE IS ' 加油类型   6- 机务加油； 10- 机务抽油 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.CREATOR IS ' 提交者 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.CREATOR_CODE IS ' 提交者⼯号 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.CREATOR_PHONE IS ' 提交者联系⽅式 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.SCENE_CONTACTS IS ' 现场联系⼈ ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.SCENE_CONTACTS_PHONE IS ' 现场联系⼈电话 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.POS IS ' 停机位 ';
COMMENT ON COLUMN FOFAS.T_AMFUEL_ORDER.CREATE_TIME IS ' 创建时间 ';
```
