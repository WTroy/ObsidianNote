title: 国内油价确认
source: 有道云笔记
imported: true
---

```sql
SELECT * FROM  T_FUEL_DOM_UNIT_PRICE t WHERE
EXPIRY_DATE =
to_date('2025-07-31', 'yyyy-mm-dd') and
ARCHIVE_FLAG = 0 and t.UPDATER is null;
update T_FUEL_DOM_UNIT_PRICE t set t.UPDATER = 'XZC', t.UPDATE_TIME =
TO_DATE('2025-07-25' , 'yyyy-MM-dd'), t.CONFIRM_STATUS= '1',
t.SECOND_UPDATER='WUST', t.SECOND_UPDATE_TIME=TO_DATE('2025-07-25' , 'yyyy-MM-dd'), t.SECOND_CONFIRM_STATUS='3' WHERE
EXPIRY_DATE =
to_date('2025-07-31', 'yyyy-mm-dd') and
ARCHIVE_FLAG = 0 and t.UPDATER is null;
```
