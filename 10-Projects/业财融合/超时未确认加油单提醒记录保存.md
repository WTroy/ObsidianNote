超时未确认加油单提醒记录保存

```sql
CREATE TABLE FOFAS.T_FUEL_REMIND_LOGS (
PROXYID VARCHAR2(100) NULL,
REFUELBILLNO VARCHAR2(50) NULL,
FLIGHTDATE VARCHAR2(100) NULL,
FLIGHTNO VARCHAR2(100) NULL,
DEPARTURECODE VARCHAR2(100) NULL,
DESTINATIONCODE VARCHAR2(100) NULL,
AIRCRAFTNO VARCHAR2(100) NULL,
RECEIVER VARCHAR2(100) NULL,
RECEIVERPHONE VARCHAR2(100) NULL,
CONTENT VARCHAR2(100) NULL,
CREATOR VARCHAR2(100) NULL,
CREATE_TIME DATE NULL,
CONSTRAINT T_FUEL_REMIND_LOGS_PK PRIMARY KEY (PROXYID)
);
COMMENT ON TABLE FOFAS.T_FUEL_REMIND_LOGS IS ' 超时未确认的加油单提醒记录表 ';
COMMENT ON COLUMN FOFAS.T_FUEL_REMIND_LOGS.PROXYID IS ' 主键 ';
COMMENT ON COLUMN FOFAS.T_FUEL_REMIND_LOGS.REFUELBILLNO IS ' 加油单号 ';
COMMENT ON COLUMN FOFAS.T_FUEL_REMIND_LOGS.FLIGHTDATE IS ' 航班⽇期 ';
COMMENT ON COLUMN FOFAS.T_FUEL_REMIND_LOGS.FLIGHTNO IS ' 航班号 ';
COMMENT ON COLUMN FOFAS.T_FUEL_REMIND_LOGS.DEPARTURECODE IS ' 出发场站 ';
COMMENT ON COLUMN FOFAS.T_FUEL_REMIND_LOGS.DESTINATIONCODE IS ' 到达场站 ';
COMMENT ON COLUMN FOFAS.T_FUEL_REMIND_LOGS.AIRCRAFTNO IS ' 机号 ';
COMMENT ON COLUMN FOFAS.T_FUEL_REMIND_LOGS.RECEIVER IS ' 接收⼈ ';
COMMENT ON COLUMN FOFAS.T_FUEL_REMIND_LOGS.RECEIVERPHONE IS ' 接收⼈⼿机号 ';
COMMENT ON COLUMN FOFAS.T_FUEL_REMIND_LOGS.CONTENT IS ' 提醒内容 ';
COMMENT ON COLUMN FOFAS.T_FUEL_REMIND_LOGS.CREATOR IS ' 创建⼈ ';
COMMENT ON COLUMN FOFAS.T_FUEL_REMIND_LOGS.CREATE_TIME IS ' 创建时间 ';
```
