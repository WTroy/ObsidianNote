预提国际邮件的存储过程更新： PKG_FUEL_FEE_ACCRUED.GET_S_FUEL_PRICE 的最下边国际邮件获取

```sql
-- 如果存在单价，则转换单价
IF (P_COUNT > 0) THEN
SELECT PRICE, CURRENCY, MEASURE_UNIT
INTO V_P_PRICES, V_P_CURRENCYS, V_P_MEASURE_UNITS
FROM (SELECT PRICE, CURRENCY, MEASURE_UNIT
FROM T_FUEL_INT_UNIT_PRICE
WHERE REFUELING_AIRPORT = V_ORI_ENG
AND V_FLIGHT_DATE >= APPLY_DATE
AND V_FLIGHT_DATE <= EXPIRY_DATE
ORDER BY APPLY_DATE DESC) A
WHERE ROWNUM = 1;
ELSE
-- 王浩
SELECT PRICE, CURRENCY, MEASURE_UNIT
INTO V_P_PRICES, V_P_CURRENCYS, V_P_MEASURE_UNITS
FROM (SELECT PRICE, CURRENCY, MEASURE_UNIT
FROM T_FUEL_INT_UNIT_PRICE
WHERE REFUELING_AIRPORT = V_ORI_ENG
ORDER BY APPLY_DATE DESC) A
WHERE ROWNUM = 1;
END IF;
```
