FROM dc_flight_plan p, T_AIRPORT A
WHERE p.ori_eng = A.AIRPORT_CODE356
AND to_char(FLIGHT_DATE, 'yyyymm') = '202507'
AND A.AREA_CODE = 'DOME'
and p.flt_status not in (' 取消 ')
AND NOT EXISTS
(SELECT *
FROM T_FUEL_DOM_UNIT_PRICE P, T_FUEL_DOM_PRICE_AIRPORT A
WHERE P.PROXYID = A.DOM_UNIT_PRICE_ID
AND A.AIRPORT = P.ORI_ENG
AND P.FLIGHT_TYPE = DECODE(P.FLIGHT_TYPE,
'D',
'DOME',
'I',
'INTL',
'RAEA',
'INTL',
P.FLIGHT_TYPE)
AND P.FLIGHT_DATE >= P.APPLY_DATE
AND (P.EXPIRY_DATE IS NULL OR P.EXPIRY_DATE >= P.FLIGHT_DATE));
-- 查询国际不存在油价的机场
SELECT DISTINCT a.cname content, a.airport_code378
FROM dc_flight_plan p, T_AIRPORT A
WHERE p.ori_eng = A.AIRPORT_CODE380
AND to_char(FLIGHT_DATE, 'yyyymm') = '202507'
AND A.AREA_CODE <> 'DOME'
and p.flt_status not in (' 取消 ')
AND NOT EXISTS
(SELECT *
FROM T_FUEL_INT_UNIT_PRICE u
WHERE u.refueling_airport = p.ori_eng
AND p.FLIGHT_DATE >= u.APPLY_DATE
AND (u.EXPIRY_DATE IS NULL OR u.EXPIRY_DATE >= P.FLIGHT_DATE));
