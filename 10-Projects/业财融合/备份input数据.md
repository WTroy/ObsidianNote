创建附件表

```sql
CREATE TABLE FOFAS.T_FUEL_INPUT_FILES (
PROXYID VARCHAR2(100) NULL,
REFUELBILLNO VARCHAR2(100) NULL,
EFS CLOB NULL
);
COMMENT ON TABLE FOFAS.T_FUEL_INPUT_FILES IS ' 实时加油数据附件信息 ';
COMMENT ON COLUMN FOFAS.T_FUEL_INPUT_FILES.PROXYID IS ' 主键，和主表保持⼀致 ';
COMMENT ON COLUMN FOFAS.T_FUEL_INPUT_FILES.REFUELBILLNO IS ' 加油单号 ';
COMMENT ON COLUMN FOFAS.T_FUEL_INPUT_FILES.EFS IS ' 附件数据 base64';
ALTER TABLE FOFAS.T_FUEL_INPUT_FILES
ADD CONSTRAINT T_FUEL_INPUT_FILES_PK PRIMARY KEY (PROXYID)
ENABLE;
```
