-- 解析完成后，将 SIA 、 CSX ， LHW,OSA,DCC 换成 XIY ， HHA ， ZGC,KIX,DCY
update t_fuel_invoice_receipt r
set r.dept_airport = 'XIY'
where r.refueling_date between to_date('2025/07/01','YYYY/MM/DD') and
to_date('2025/07/28','YYYY/MM/DD')
and r.dept_airport = 'SIA';
update t_fuel_invoice_receipt r
set r.dept_airport = 'HHA'
where r.refueling_date between to_date('2025/07/01','YYYY/MM/DD') and
to_date('2025/07/28','YYYY/MM/DD')
and r.dept_airport = 'CSX';
update t_fuel_invoice_receipt r
set r.dept_airport = 'ZGC'
where r.refueling_date between to_date('2025/07/01','YYYY/MM/DD') and
to_date('2025/07/28','YYYY/MM/DD')
and r.dept_airport = 'LHW';
update t_fuel_invoice_receipt r
set r.dept_airport = 'KIX'
where r.refueling_date between to_date('2025/07/01','YYYY/MM/DD') and
to_date('2025/07/28','YYYY/MM/DD')
and r.dept_airport = 'OSA';
update t_fuel_invoice_receipt r
set r.dept_airport = 'DCY'
where r.refueling_date between to_date('2025/07/01','YYYY/MM/DD') and
to_date('2025/07/28','YYYY/MM/DD')
and r.dept_airport = 'DCC';
commit;

update t_fuel_invoice_receipt r

commit;
-- 去除航班号中的 -57
update t_fuel_invoice_receipt r
set r.flight_no = trim(replace(r.flight_no, '-', ''))
where r.refueling_date between to_date('2023/11/01','YYYY/MM/DD') and
to_date('2025/07/28','YYYY/MM/DD')
and r.flight_no like '%-%';
commit;
-- 去除机号中的 -66
update t_fuel_invoice_receipt r
set r.tail_no = trim(replace(r.tail_no, '-', ''))
where r.refueling_date between to_date('2023/11/01','YYYY/MM/DD') and
to_date('2025/07/28','YYYY/MM/DD')
and r.tail_no like '%-%';
commit;
