DOCKER

```bash
# 创建 linux 容器
docker run -p 8888:8888 -it -d rockylinux:8.92
# 从命令⾏进⼊容器
docker exec -it 容器 id bash
# 安装  Miniconda https://docs.anaconda.com/free/miniconda/
# 以 Intel 处理器为例，⽂件名可能会更改
sh Miniconda3-py39_4.12.0-MacOSX-x86_64.sh -b
# 初始化终端 Shell
~/miniconda3/bin/conda init
# ⽤下⾯的命令创建⼀个新的环境
conda create --name d2l python=3.9 -y
# 激活环境
conda activate d2l
# 安装 tensorflow 和 d2l 软件包
pip install tensorflow==2.8.015
pip install tensorflow-probability==0.16.016
pip install d2l==0.17.617
# 下载对应代码
mkdir d2l-zh && cd d2l-zh
curl https://zh-v2.d2l.ai/d2l-zh-2.0.0.zip -o d2l-zh.zip
unzip d2l-zh.zip && rm d2l-zh.zip
cd tensorflow
# 启动 jupyter
jupyter notebook --allow-root --ip=*
# jupyter 提示功能插件
pip install jupyter_contrib_nbextensions
jupyter contrib nbextension install --user --skip-running-check
# 重启 jupyter
# 导包 tensorflow 失败：执⾏  pip install protobuf==3.19.0 然后重启 jupyter
jupyter notebook stop
# 在运⾏书籍代码、更新深度学习框架或 d2l 软件包之前，请始终执⾏  conda activate d2l 以激活运⾏时
```
